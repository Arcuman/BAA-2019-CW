integer function Inlength(integer num)
{
type integer len = 0;
type integer tempNum = num;
type integer degree = 1;
while: tempNum ! 0
do
[
 tempNum = num / power(10,degree);
 degree = degree + 1;
]
len = degree - 1;
return len;
}

{

 write posit;

return 0;
 ]

return (-1);

return (-1);
}
