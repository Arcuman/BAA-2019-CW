integer function outeven(integer a , integer b)
{
type integer temp;
write "Even num from: ";
write a;
write " to ";
write b;
writeline ":";
b = b +1;
while a < b
do
[
temp = a % 2;
is temp & 0
true[write a;write " ";]
a = a + 1;
]
writeline " ";
return 0;
}
